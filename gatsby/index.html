2. 데이터 수집 함수 구현

def get_movie_reviews(movie_code, pages=10):
    reviews = []

    for page in range(1, pages + 1):
        url = f"https://movie.naver.com/movie/point/af/list.naver?st=mcode&target=after&sword={movie_code}&page={page}"
        response = requests.get(url)
        soup = BeautifulSoup(response.text, 'html.parser')

        items = soup.select('table.list_netizen > tbody > tr')

        for item in items:
            try:
                score = item.select_one('td.title em').text
                review = item.select_one('td.title a')['title']
                date = item.select_one('td.num').text.strip()

                reviews.append({
                    'score': int(score),
                    'review': review,
                    'date': date
                })
            except:
                continue

        print(f"페이지 {page} 수집 완료")

    return pd.DataFrame(reviews)

# 데이터 수집 실행
movie_code = "161967"  # 기생충 영화 코드
df = get_movie_reviews(movie_code)

3. 데이터 기본 분석

# 기본 통계
print("\n=== 기본 통계 ===")
print("전체 리뷰 수:", len(df))
print("평점 평균:", round(df['score'].mean(), 2))
print("평점 중앙값:", df['score'].median())
print("평점 표준편차:", round(df['score'].std(), 2))

# 평점 분포
print("\n=== 평점 분포 ===")
score_dist = df['score'].value_counts().sort_index()
print(score_dist)

# 날짜별 리뷰 수
df['date'] = pd.to_datetime(df['date'])
daily_reviews = df.groupby(df['date'].dt.date).size()

4. 데이터 시각화

# 1. 평점 분포 히스토그램

5. 워드클라우드 생성

6. 간단한 감성 분석

# 긍정/부정 리뷰 분류 (간단한 버전)
def classify_sentiment(score):
    if score >= 8:
        return '긍정'
    elif score <= 4:
        return '부정'
    else:
        return '중립'

df['sentiment'] = df['score'].apply(classify_sentiment)

# 감성 분포 시각화
plt.figure(figsize=(8, 6))
sns.countplot(data=df, x='sentiment')
plt.title('리뷰 감성 분포')
plt.show()

# 감성별 대표 리뷰 출력
print("\n=== 감성별 대표 리뷰 ===")
for sentiment in ['긍정', '중립', '부정']:
    print(f"\n{sentiment} 리뷰 예시:")
    sample_reviews = df[df['sentiment'] == sentiment]['review'].sample(n=3)
    for idx, review in enumerate(sample_reviews, 1):
        print(f"{idx}. {review}")

이 코드는 네이버 영화의 어떤 영화에 대해서도 적용할 수 있으며, movie_code만 변경하면 됩니다. 영화 코드는 네이버 영화 URL에서 확인할 수 있습니다.
